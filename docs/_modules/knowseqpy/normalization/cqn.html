def fix_predictor(zz, subindex, varname="B"):
    zz_fit = zz[subindex]
    knots = np.quantile(zz_fit, [0.025, 0.25, 0.50, 0.75, 0.975]) + np.array([+0.01, 0, 0, 0, -0.01])
    grid = np.linspace(min(zz_fit), max(zz_fit), 101)
    zz_out = zz.copy()
    zz2 = zz[~subindex]
    zz2[zz2 < min(zz_fit)] = min(zz_fit)
    zz2[zz2 > max(zz_fit)] = max(zz_fit)
    zz_out[~subindex] = zz2
    return {"fit": zz_fit, "out": zz_out, "knots": knots, "grid": grid}

def mix_qn(y0, ctrl_id, NQ, mix_param, max_q=0.95, min_q=0.01, LTrule=1):
    ECDF = np.sort(y0[ctrl_id])
    Q = np.searchsorted(ECDF, y0, side='right') / len(ECDF)

    id0 = np.where(Q < min_q)[0]
    id2 = np.where(Q > max_q)[0]
    B = len(id2)
    Q2 = max_q + (1 - max_q) * (np.arange(1, B + 1) / (B + 1))
    y2o = np.argsort(y0[id2])
    Q[id2[y2o]] = Q2

    ynorm = np.zeros_like(y0)
    ynorm[id0] = y0[id0] - np.quantile(y0[ctrl_id], min_q) + np.quantile(NQ, min_q)
    ynorm[np.setdiff1d(np.arange(len(y0)), np.concatenate([id0, id2]))] = np.interp(
        Q[np.setdiff1d(np.arange(len(y0)), np.concatenate([id0, id2]))], np.linspace(0, 1, len(NQ)), np.sort(NQ))
    ynorm[id2] = np.interp(Q[id2], np.linspace(0, 1, len(mix_param['means'])), mix_param['means'])

    return ynorm

def SQN2(y, N_mix=5, ctrl_id=None, model_weight=0.9, max_q=0.95, min_q=0.01, LTrule=1):
    QE = np.apply_along_axis(np.sort, 0, y[ctrl_id])
    QN = np.median(QE, axis=1)

    gmm = GaussianMixture(n_components=N_mix).fit(QN.reshape(-1, 1))
    mix_param = {
        'means': gmm.means_.flatten(),
        'covariances': gmm.covariances_.flatten(),
        'weights': gmm.weights_
    }

    qq = np.linspace(1 / (2 * len(QN)), 1 - 1 / (2 * len(QN)), len(QN))
    qq = norm.ppf(qq, loc=np.dot(mix_param['weights'], mix_param['means']),
                  scale=np.sqrt(np.dot(mix_param['weights'], mix_param['covariances'])))
    QN1 = QN * (1 - model_weight) + qq * model_weight

    ynorm = np.apply_along_axis(mix_qn, 0, y, ctrl_id=ctrl_id, NQ=QN1, mix_param=mix_param, max_q=max_q, min_q=min_q,
                                LTrule=LTrule)
    model_par = {"N_mix": N_mix, "model_weight": model_weight, "max_q": max_q, "min_q": min_q, "LTrule": LTrule}

    return {"yout": ynorm, "QN": QN1, "par": {"mix_param": mix_param, "model_par": model_par}}

def cqn(counts: pd.DataFrame, x: pd.Series, lengths: pd.Series, size_factors: pd.Series = None,
        subindex: pd.Series = None,
        tau: float = 0.5, sqn: bool = True, length_method: str = "smooth", verbose: bool = False) -> dict:
    length_method = length_method.lower()
    if length_method not in ["smooth", "fixed"]:
        raise ValueError("lengthMethod must be either 'smooth' or 'fixed'")

    if len(lengths) == 1:
        lengths = np.repeat(lengths, counts.shape[0])
    if counts.ndim == 1:
        counts = counts[:, None]
    if counts.shape[0] != len(x) or counts.shape[0] != len(lengths):
        raise ValueError(
            "arguments 'counts' need to have the same number of rows as the length of arguments 'x' and 'lengths'")
    if any(lengths <= 0):
        raise ValueError("argument 'lengths' need to be greater than zero")
    if size_factors is not None and len(size_factors) != counts.shape[1]:
        raise ValueError(
            "argument 'sizeFactors' (when used) needs to have the same length as the number of columns of argument 'counts'")
    if subindex is not None and (min(subindex) <= 0 or max(subindex) > counts.shape[0]):
        raise ValueError(
            "argument 'subindex' (when used) needs to be indices into the number of rows of argument 'counts'")

    if size_factors is None:
        size_factors = counts.sum(axis=0)

    if subindex is None:
        subindex = counts.mean(axis=1) > 50

    y = np.log2(counts + 1) - np.log2(size_factors / 10 ** 6)
    if length_method == "fixed":
        y -= np.log2(lengths / 1000)[:, None]
    yfit = y[subindex]

    x1p = fix_predictor(x, subindex)
    x1_knots = x1p["knots"]

    if length_method == "smooth":
        x2p = fix_predictor(np.log2(lengths / 1000), subindex)
        x2_knots = x2p["knots"]
        df_fit = pd.DataFrame({"x1": x1p["fit"], "x2": x2p["fit"]})
        df_predict = pd.DataFrame({"x1": x1p["out"], "x2": x2p["out"]})
        df_func = pd.concat([pd.DataFrame({"x1": x1p["grid"], "x2": [np.median(x2p["grid"])] * len(x1p["grid"])}),
                             pd.DataFrame({"x1": [np.median(x1p["grid"])] * len(x2p["grid"]), "x2": x2p["grid"]})],
                            ignore_index=True)
    else:
        x2p = None
        df_fit = pd.DataFrame({"x1": x1p["fit"]})
        df_predict = pd.DataFrame({"x1": x1p["out"]})
        df_func = pd.DataFrame({"x1": x1p["grid"]})

    if verbose:
        print("RQ fit ", end="")
    regr = []
    for ii in range(yfit.shape[1]):
        if verbose:
            print(".", end="")
        df_fit["y"] = yfit[:, ii]
        if length_method == "smooth":
            fit = QuantReg(df_fit["y"], df_fit[["x1", "x2"]]).fit(q=tau)
        else:
            fit = QuantReg(df_fit["y"], df_fit[["x1"]]).fit(q=tau)
        fitted = fit.predict(df_predict)
        func = fit.predict(df_func)
        regr.append({"fitted": fitted, "func": func, "coef": fit.params})
    if verbose:
        print()

    fitted = np.column_stack([r["fitted"] for r in regr])
    func = np.column_stack([r["func"] for r in regr])

    k = np.argsort(x[subindex])[len(subindex) // 2]
    offset0 = np.median(fitted[subindex[k]])

    residuals = y - fitted
    if sqn:
        if verbose:
            print("SQN ", end="")
        sqn_result = SQN2(residuals, ctrl_id=subindex)
        residuals_sqn = sqn_result["yout"]
        if verbose:
            print(".")
        offset = residuals_sqn + offset0 - y
    else:
        offset = residuals + offset0 - y

    glm_offset = (-offset) - np.log2(size_factors / 10 ** 6)
    glm_offset *= np.log(2)

    result = {
        "counts": counts,
        "lengths": lengths,
        "sizeFactors": size_factors,
        "subindex": subindex,
        "y": y,
        "x": x,
        "offset": offset,
        "offset0": offset0,
        "glm_offset": glm_offset,
        "func1": func[:, :len(x1p["grid"])],
        "func2": func[:, len(x1p["grid"]):] if length_method == "smooth" else None,
        "grid1": x1p["grid"],
        "grid2": x2p["grid"] if length_method == "smooth" else None,
        "knots1": x1p["knots"],
        "knots2": x2p["knots"] if length_method == "smooth" else None,
        "call": "cqn"
    }

    return result
